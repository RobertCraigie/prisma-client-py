22class OSAgnosticSingleFileExtension(SingleFileSnapshotExtension): 

23 # syrupy's types are only written to target mypy, as such 

24 # pyright does not understand them and reports them as unknown. 

25 # As this method is only called internally it is safe to type as Any 

26 @override 

27 def serialize( 

28 self, 

29 data: Any, 

30 *, 

31 exclude: Optional[Any] = None, 

32 matcher: Optional[Any] = None, 

33 ) -> bytes: 

34 serialized = DataSerializer.serialize(data, exclude=exclude, matcher=matcher) 

35 return bytes(serialized, 'utf-8')

59def get_files_from_templates(directory: Path) -> List[str]: 

60 """Return a list of all auto-generated python modules""" 

61 files: List[str] = [] 

62 

63 for template in directory.iterdir(): 

64 if template.is_dir(): 

65 files.extend(get_files_from_templates(template)) 

66 elif template.name.endswith('.py.jinja') and not template.name.startswith('_'): 

67 if directory.name == 'templates': 

68 name = template.name 

69 else: 

70 name = str(template.relative_to(template.parent.parent)) 

71 

72 files.append(remove_suffix(name, '.jinja').replace('\\', '/')) 

73 

74 return files
